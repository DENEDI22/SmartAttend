**`Dockerfile`** (per FOUND-06, Research multi-arch section):
```dockerfile
# syntax=docker/dockerfile:1
FROM python:3.11-slim

WORKDIR /app

# Install dependencies first (cached layer — only rebuilds when requirements.txt changes)
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Copy application code
COPY app/ ./app/

# Create data directory for SQLite database persistence
RUN mkdir -p /app/data

CMD ["uvicorn", "app.main:app", "--host", "0.0.0.0", "--port", "8000"]
```

NOTE: `python:3.11-slim` is natively multi-arch on Docker Hub (amd64, arm64, arm/v7) — no arch-specific logic needed in the Dockerfile. Multi-arch manifest is produced at build time by `docker buildx` (Research Pattern / note on multi-arch build).

**`mosquitto/config/mosquitto.conf`** (per FOUND-07, Research Pattern 6):
```ini
# Mosquitto 2.0+ requires explicit listener + allow_anonymous.
# Without these two lines, the broker only binds to 127.0.0.1 inside
# its own container — other Docker services cannot connect.
listener 1883
allow_anonymous true
persistence true
persistence_location /mosquitto/data/
log_dest stdout
```

CRITICAL: Do NOT omit `listener 1883` — Mosquitto 2.0 changed the default to local-only, causing confusing "connection refused" errors from app and dummy client containers (Research Pitfall 1).

**`docker-compose.yml`** (per FOUND-05, Research Pattern 7 — 5 services: server, mqtt, client-e101/e102/e103):
```yaml
version: "3.9"

services:
  server:
    build: .
    ports:
      - "8000:8000"
    env_file: .env
    depends_on:
      - mqtt
    volumes:
      - ./data:/app/data    # SQLite database persists across restarts
    restart: unless-stopped

  mqtt:
    image: eclipse-mosquitto:2
    ports:
      - "1883:1883"
    volumes:
      - ./mosquitto/config:/mosquitto/config:ro
      - mosquitto-data:/mosquitto/data
    restart: unless-stopped

  client-e101:
    build: ./dummy_client
    environment:
      - DEVICE_ID=e101
      - ROOM=E101
      - MQTT_BROKER=mqtt
      - MQTT_PORT=1883
      - LUX_VALUE=400
    depends_on:
      - mqtt
    restart: unless-stopped

  client-e102:
    build: ./dummy_client
    environment:
      - DEVICE_ID=e102
      - ROOM=E102
      - MQTT_BROKER=mqtt
      - MQTT_PORT=1883
      - LUX_VALUE=350
    depends_on:
      - mqtt
    restart: unless-stopped

  client-e103:
    build: ./dummy_client
    environment:
      - DEVICE_ID=e103
      - ROOM=E103
      - MQTT_BROKER=mqtt
      - MQTT_PORT=1883
      - LUX_VALUE=500
    depends_on:
      - mqtt
    restart: unless-stopped

volumes:
  mosquitto-data:
```

NOTE: Database path in docker-compose uses bind mount `./data:/app/data` — DATABASE_URL in .env must use `sqlite:////app/data/smartattend.db` (4 slashes = absolute path inside container). See Research open question 3.
