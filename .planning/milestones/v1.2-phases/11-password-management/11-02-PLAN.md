    **Step 3: Add inline `<script>` at end of template** (before `{% endblock %}`):
    
    ```javascript
    <script>
    function openResetDialog(userId, userName) {
      const dialog = document.getElementById('reset-dialog');
      const form = document.getElementById('reset-form');
      const subtitle = document.getElementById('reset-dialog-subtitle');
      const input = document.getElementById('reset-password-input');
      const hint = document.getElementById('generated-hint');
      
      form.action = '/admin/users/' + userId + '/reset-password';
      subtitle.textContent = userName;
      input.value = '';
      hint.style.display = 'none';
      dialog.showModal();
    }
    
    function generatePassword() {
      const chars = 'ABCDEFGHJKLMNPQRSTUVWXYZabcdefghjkmnpqrstuvwxyz23456789';
      let pwd = '';
      const array = new Uint8Array(12);
      crypto.getRandomValues(array);
      for (let i = 0; i < 12; i++) {
        pwd += chars[array[i] % chars.length];
      }
      document.getElementById('reset-password-input').value = pwd;
      document.getElementById('generated-hint').style.display = 'inline';
    }
    </script>
    ```
    
    Auto-generated password: 12 chars, alphanumeric, no ambiguous chars (0/O, 1/l/I removed from charset). Uses `crypto.getRandomValues()` for randomness. Displayed in the text input (type="text" so admin can see and copy it).
